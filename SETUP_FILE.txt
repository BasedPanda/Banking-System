create table Customer_Info (Customer_Id char(5) primary key,Name varchar(30) not null,Phone_Number char(10) not null,Monthly_Income int  not null,Credit_Score float not null);
create table Savings(Account_Number char(6)primary key, Customer_Id char(5) not null, Customer_Name varchar(30) not null, Amount int not null, Interest_Rate float not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table current(Account_Number char(6) primary key, Customer_Id char(5) not null, Customer_Name varchar(30) not null, Amount int not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table fixed_deposit(Account_number char(6) primary key, Customer_Id char(5) not null, Customer_Name varchar(30) not null, Amount int not null, Start_date date not null, duration_in_months int not null, End_Date date not null, Interest_Rate float not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table Home_Loan (Account_number char(6) primary key, Customer_Id char(5) not null, Customer_Name varchar(30) not null, Loan_Amount int not null, Interest_Rate float not null,Start_Date date  not null, End_Date date  not null, EMI int  not null, Balance int not null,Paid_this_month varchar(3) not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table Education_Loan (Account_number char(6) primary key, Customer_Id char(5) not null, Customer_Name varchar(30) not null, Loan_Amount int not null, Interest_Rate float not null,Education_Period_in_months int not null,Start_Date date  not null, End_Date date  not null, EMI int  not null,Balance int not null,Paid_this_month varchar(3) not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table Transactions(Transaction_Id char(7) primary key, Account_number char(6) not null, Customer_Id char(5) not null,Customer_Name varchar(30) not null,Account_Type varchar(30) not null,Amount int not null, Date date not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
create table Fine (Default_Id char(8) primary key, Customer_Id char(5) not null,Customer_Name varchar(30) not null, Account_Number char(6) not null, Loan_Type varchar(30) not null, Fine_Value int not null, Default_Number int not null,Date_of_Fine date not null, foreign key(Customer_Id) references Customer_Info(Customer_Id));
